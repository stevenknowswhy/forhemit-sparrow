pub async fn research(&self, fingerprint: ProductFingerprint) -> ComparisonReport {
// 1. Build anonymized search queries
let queries = self.build_queries(&fingerprint);
// 2. Dispatch parallel searches across sources
let results = join_all![
self.search_amazon(&queries),
self.search_google_shopping(&queries),
self.search_ebay(&queries),  // if refurbished enabled
self.search_specialized(&queries),
].await;

.flat_map(|r| self.extract_offers(r))
.collect();

.map(|o| self.score_offer(o, &fingerprint))
.filter(|s| s.comparison_score >= 65)  // minimum viable
.collect();
// 5. Generate comparison report
ComparisonReport::new(scored)
}
